	LIST	P=16F84
	INCLUDE	"P16F84.INC"
	
	CLRF	PORTB			; PORTB VE PORTA SIFIRLANIR
	CLRF	PORTA
	BSF	STATUS,5
	CLRF	TRISB
	CLRF	PORTB
	MOVLW	B'00000000'
	MOVWF	TRISB
	BCF	STATUS,5
	

BUTON
	BTFSS	PORTA,1
	GOTO	ONLAR
	NOP
	BTFSS	PORTA,0
	GOTO	BIRLER
	NOP
	GOTO	BUTON

;OGRENCI NUMARAM B1810.032033 OLDUGU ICIN IKI BUTONDA DA 3 GORUNECEK
;PORTA 0 ISE BIRLER ALGORTIMASI 1 ISE ONLAR ALGORITMASINA GIDER

ONLAR			
	MOVLW	h'4F'
	MOVWF	PORTB
	CALL	RAKAMLAR
	GOTO	BUTON
	
BIRLER
	MOVLW	h'4F'
	MOVWF	PORTB
	CALL	RAKAMLAR
	GOTO	BUTON

		
RAKAMLAR
		ADDWF	PCL,F	 ;OFSET DEGERINI PCL E EKLE
		RETLW	0X3F	    ;0 ICIN LED SURUCU
		RETLW	0X06	    ;1 ICIN LED SURUCU
		RETLW	0X05	    ;2 ICIN LED SURUCU
		RETLW	0X4F	    ;3 ICIN LED SURUCU
		RETLW	0X66	    ;4 ICIN LED SURUCU
		RETLW	0X6D	    ;5 ICIN LED SURUCU
		RETLW	0X7D	    ;6 ICIN LED SURUCU
		RETLW	0X07	    ;7 ICIN LED SURUCU
		RETLW	0X7F	    ;8 ICIN LED SURUCU
		RETLW	0X6F	    ;9 ICIN LED SURUCU
		RETLW	0X77	    ;A ICIN LED SURUCU
		RETLW	0X7C	    ;B ICIN LED SURUCU
		RETLW	0X39	    ;C ICIN LED SURUCU
		RETLW	0X5E	    ;D ICIN LED SURUCU
		RETLW	0X79	    ;E ICIN LED SURUCU
		RETLW	0X71	    ;F ICIN LED SURUCU
		RETLW	0X80	    ;. ICIN LED SURUCU
		RETURN

	END
